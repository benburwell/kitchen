ingredients:
  - item: chicken bouillon
    qty: 1
    unit: packet
  - item: pastina
    qty: 0.3
    unit: c
  - item: butter
  - item: milk     
procedure:
  - Boil 1qt water with 1 packet chicken bouillon. 
  - Add 0.3c pastina and cook 5-7 minutes, stirring.
  - Drain if needed. Add butter and a splash of milk.
